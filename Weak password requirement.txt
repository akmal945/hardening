Prevent use old password

<?php
// Get submitted old and new passwords
$oldPassword = $_POST['old_password'];
$newPassword = $_POST['new_password'];

// ğŸ”„ Fetch current user data (e.g., from DB)

// âœ… Step 1: Verify old password matches stored password

    die("Old password is incorrect.");
}

// âŒ Step 2: Prevent reuse â€” new password must NOT match the current one
if (password_verify($newPassword, $user['hashed_password'])) {
    die("New password cannot be the same as the old password.");
}

// âœ… Step 3: Hash and save the new password
$newHashedPassword = password_hash($newPassword, PASSWORD_DEFAULT);
